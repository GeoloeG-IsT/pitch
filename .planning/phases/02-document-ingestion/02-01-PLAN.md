```sql
CREATE TABLE public.documents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  file_name TEXT NOT NULL,
  file_type TEXT NOT NULL CHECK (file_type IN ('pdf', 'xlsx', 'md', 'txt')),
  file_size_bytes BIGINT,
  status TEXT NOT NULL DEFAULT 'pending' CHECK (status IN ('pending', 'processing', 'ready', 'error')),
  metadata JSONB DEFAULT '{}',
  view_count INT DEFAULT 0,
  last_viewed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE public.chunks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  document_id UUID NOT NULL REFERENCES public.documents(id) ON DELETE CASCADE,
  content TEXT NOT NULL,
  embedding vector(1536),
  section_number INT,
  page_number INT,
  chunk_type TEXT NOT NULL DEFAULT 'text' CHECK (chunk_type IN ('text', 'table', 'heading', 'image_caption')),
  metadata JSONB DEFAULT '{}',
  token_count INT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

1. Create `apps/api/app/services/node_mapper.py`:
   - `import tiktoken` and cache encoding: `_encoding = tiktoken.encoding_for_model("text-embedding-3-small")`
   - Function `node_to_chunk_record(node: BaseNode, document_id: str) -> dict`:
     - Extract `page_label` from metadata, convert to int as `page_number`
     - Determine `chunk_type`: if metadata has `header_path` or `heading` -> "heading", else check for markdown table syntax (pipes) in content -> "table", else "text"
     - Count tokens with `len(_encoding.encode(text))`
     - Return dict with keys: document_id, content, embedding, section_number (defaults to page_number), page_number, chunk_type, metadata (filtered: exclude page_label, file_path, file_name, chunk_type keys), token_count
   - Import `BaseNode` from `llama_index.core.schema`

2. Create `apps/api/app/services/parsers/pdf_pipeline.py`:
   - Function `run_pdf_pipeline(file_bytes: bytes) -> list[BaseNode]`:
     - Write bytes to tempfile with `.pdf` suffix
     - Load with `PyMuPDFReader().load_data(temp_path)` -- returns one Document per page
     - Create IngestionPipeline with transformations: `[SentenceSplitter(chunk_size=500, chunk_overlap=50), OpenAIEmbedding(model="text-embedding-3-small")]`
     - NOTE: Do NOT use MarkdownNodeParser for PDFs (PyMuPDFReader already splits per page)
     - Run `await pipeline.arun(documents=documents)`
     - Clean up temp file in finally block
     - Return list of nodes
   - Make function async

3. Create `apps/api/app/services/parsers/markdown_pipeline.py`:
   - Function `run_markdown_pipeline(text: str) -> list[BaseNode]`:
     - Create `Document(text=text)` from `llama_index.core`
     - Create IngestionPipeline with transformations: `[MarkdownNodeParser(), SentenceSplitter(chunk_size=500, chunk_overlap=50), OpenAIEmbedding(model="text-embedding-3-small")]`
     - Run `await pipeline.arun(documents=[doc])`
     - Return list of nodes
   - Make function async

4. Create `apps/api/app/services/parsers/excel_parser.py`:
   - Function `parse_excel(file_bytes: bytes) -> list[dict]`:
     - Load workbook with `openpyxl.load_workbook(io.BytesIO(file_bytes), read_only=True, data_only=True)`
     - For each sheet: extract all rows as list of lists (stringify cells, None -> "")
     - Skip empty sheets (all cells empty)
     - Generate LLM summary via OpenAI gpt-4o-mini: system prompt = "Summarize this financial spreadsheet data in clear natural language. Include all key numbers, metrics, and relationships. Be comprehensive -- this summary will be used for semantic search."
     - Generate embedding via `OpenAIEmbedding(model="text-embedding-3-small").aget_text_embedding(summary)`
     - Count tokens with tiktoken
     - Return list of dicts with keys: content (summary), embedding, section_number (sheet index starting at 1), page_number (None), chunk_type ("table"), metadata ({sheet_name, raw_data (rows), row_count, col_count}), token_count
   - Helper `_rows_to_text(rows) -> str`: format rows as tab-separated text with row numbers
   - Helper `_generate_summary(sheet_name, table_text) -> str`: call gpt-4o-mini
   - Make function async

5. Create `apps/api/app/services/pipeline.py`:
   - Import all three parsers
   - Import node_mapper
   - Function `process_file(file_bytes: bytes, file_type: str, document_id: str) -> list[dict]`:
     - If file_type == "xlsx": call `parse_excel(file_bytes)`, set document_id on each chunk dict, return
     - If file_type == "pdf": call `run_pdf_pipeline(file_bytes)`, map each node through `node_to_chunk_record(node, document_id)`, return
     - If file_type in ("md", "txt"): call `run_markdown_pipeline(file_bytes.decode("utf-8"))`, map each node through `node_to_chunk_record(node, document_id)`, return
     - Raise ValueError for unsupported file types
   - Make function async

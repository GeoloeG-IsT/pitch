   ```sql
   -- Live session tracking
   CREATE TABLE public.live_sessions (
     id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
     founder_id UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
     started_at TIMESTAMPTZ NOT NULL DEFAULT now(),
     ended_at TIMESTAMPTZ  -- NULL means active
   );

   CREATE INDEX idx_live_sessions_active ON public.live_sessions(founder_id)
     WHERE ended_at IS NULL;

   ALTER TABLE public.live_sessions ENABLE ROW LEVEL SECURITY;

   CREATE POLICY "Founders manage own sessions"
     ON public.live_sessions FOR ALL
     USING (auth.uid() = founder_id);

   -- Add live_session_id to queries for tracking which questions came during live sessions
   ALTER TABLE public.queries ADD COLUMN IF NOT EXISTS live_session_id UUID
     REFERENCES public.live_sessions(id);

   -- Expand review_status CHECK to include 'dismissed'
   ALTER TABLE public.queries DROP CONSTRAINT IF EXISTS queries_review_status_check;
   ALTER TABLE public.queries ADD CONSTRAINT queries_review_status_check
     CHECK (review_status IN ('auto_published', 'pending_review', 'approved', 'edited', 'rejected', 'dismissed'));
   ```

1. **Create `apps/api/app/api/v1/sessions.py`** with:

   In-memory session cache:
   ```python
   _active_sessions: dict[str, dict] = {}  # founder_id -> {"id": session_id, "started_at": ...}
   ```

   Helper function `get_active_session(founder_id: str) -> dict | None` that returns the cached session or None. Also `get_active_session_by_id(session_id: str) -> dict | None` to find by session ID.

   Endpoints:
   - `POST /sessions` (auth required via `Depends(get_current_user)`):
     - Check if founder already has an active session in `_active_sessions`. If yes, return 409 Conflict.
     - Insert into `live_sessions` table with `founder_id = user["sub"]`.
     - Update `_active_sessions` cache.
     - Call `broadcast_session_event(user["sub"], {"type": "session_started"})` to notify connected investors.
     - Return `SessionResponse`.

   - `DELETE /sessions/{session_id}` (auth required):
     - Fetch session from DB, verify `founder_id == user["sub"]` and `ended_at IS NULL`.
     - Update `ended_at = now()` in DB.
     - Remove from `_active_sessions` cache.
     - Call `broadcast_session_event(user["sub"], {"type": "session_ended"})`.
     - Return `SessionResponse` with `is_active=False`.

   - `GET /sessions/active` (auth required):
     - Check `_active_sessions` for `user["sub"]`.
     - If found, return `SessionResponse`. If not, return `{"session": null}`.

   - `PUT /sessions/{session_id}/questions/{query_id}` (auth required):
     - Accept `SessionAction` body.
     - Fetch query from DB, verify it has `live_session_id == session_id`.
     - Handle actions:
       - `"approve"`: Set `review_status = "approved"`, `status = "complete"`. Use existing answer as-is.
       - `"edit"`: Set `review_status = "edited"`, `status = "complete"`, `founder_answer = body.edited_answer`, `answer = body.edited_answer`.
       - `"override"`: Set `review_status = "edited"`, `status = "complete"`, `founder_answer = body.edited_answer`, `answer = body.edited_answer`.
       - `"dismiss"`: Set `review_status = "dismissed"`, `status = "complete"`.
     - Update DB with `reviewed_by = user["sub"]`, `reviewed_at = now()`.
     - For approve/edit/override: call `broadcast_approved_answer(query_id, answer_text, "verified")`.
     - For dismiss: call `broadcast_dismissed_question(query_id)` (new function in notifications.py).
     - Return updated `ReviewItem`.

   On module load: add a startup function `load_active_sessions()` that queries `live_sessions WHERE ended_at IS NULL` and populates `_active_sessions`. Register this via FastAPI `@router.on_event("startup")` or as a plain async function called from main.py.
